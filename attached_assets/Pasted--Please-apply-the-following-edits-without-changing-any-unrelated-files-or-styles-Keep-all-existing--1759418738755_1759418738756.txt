/components/SiteLogo.tsx (add this new component)

"use client";

import Image from "next/image";
import Link from "next/link";
import { usePathname } from "next/navigation";
import clsx from "clsx";

type Props = {
  className?: string;
  size?: number; // height in px
  glow?: boolean;
  ariaLabel?: string;
};

export default function SiteLogo({
  className,
  size = 24,
  glow = true,
  ariaLabel = "Lendgismo Home",
}: Props) {
  const pathname = usePathname();

  // Prefer SVGs; fall back to PNGs if SVG not present (Next will handle).
  const lightSrc = "/brand/logo-light.svg"; // for dark header/backgrounds
  const darkSrc  = "/brand/logo-dark.svg";  // for light backgrounds

  const height = size;
  const width = Math.round(size * 5.2); // adjust aspect ratio to your logo (w:h ~5.2:1 is common)
  const base = "inline-flex items-center select-none";

  return (
    <Link
      href="/"
      aria-label={ariaLabel}
      className={clsx(base, className)}
      prefetch={false}
    >
      {/* Dark header version (logo appears light) */}
      <span
        className={clsx(
          "block dark:inline",
          "hidden",
          glow && "lglogo-glow"
        )}
        style={{ lineHeight: 0 }}
      >
        <Image
          src={lightSrc}
          alt="Lendgismo"
          width={width}
          height={height}
          priority={pathname === "/"}
        />
      </span>

      {/* Light header version (logo appears dark) */}
      <span className="block dark:hidden" style={{ lineHeight: 0 }}>
        <Image
          src={darkSrc}
          alt="Lendgismo"
          width={width}
          height={height}
          priority={pathname === "/"}
        />
      </span>
    </Link>
  );
}
